CREATE TABLE orders (
	order_id INT AUTO_INCREMENT PRIMARY KEY, -- 입력하지 않으면 1씩 증가하여 저장
	customer_id INT NOT NULL,
	product_id INT NOT NULL,
	quantity INT NOT NULL,
	order_date DATETIME DEFAULT CURRENT_TIMESTAMP, -- 현재시간을 저장
		status VARCHAR(20) NOT NULL DEFAULT '주문접수', -- 초기설정값

);

CREATE TABLE products (
  product_id  INT AUTO_INCREMENT PRIMARY KEY,
  product_name  VARCHAR(100) NOT NULL,
  product_code  VARCHAR(20) UNIQUE,
  price INT NOT NULL,
  stock_count INT NOT NULL DEFAULT 0
);

CREATE TABLE customers (
  customer_id INT AUTO_INCREMENT PRIMARY KEY,
  name  VARCHAR(50) NOT NULL
);

  order_id  INT AUTO_INCREMENT PRIMARY KEY,
  customer_id INT NOT NULL,
  order_date  DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT  fk_orders_customers FOREIGN KEY (customer_id)

);

SELECT customer_name, SUM(price * quantity) AS `총 구매 금액`
